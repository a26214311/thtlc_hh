script_enemy_main
{//逃げながら撃つ、いや、撃って逃げるだ！
	let count=0;
	let cx=GetCenterX();//STGシーンの中心x座標を取得
	let csd=GetCurrentScriptDirectory;
	let imgEnemy;
	alternative(GetArgument)
		case(0,1){imgEnemy=csd~".\img\fairy_blue.png";}	//青妖精
		case(2,3){imgEnemy=csd~".\img\fairy_red.png";}	//赤妖精
	let imgAngle=0;
	let shotAngle=0;
	let ShotSE = GetCurrentScriptDirectory~"se\shot1.wav";
	let ShotSE2= GetCurrentScriptDirectory~"se\shot2.wav";

	let MinX = GetClipMinX;
	let MaxX = GetClipMaxX;
	let MinY = GetClipMinY;
	let MaxY = GetClipMaxY;

	// アイテムを落とす数
	let ItemNam_1up=0;
	let ItemNam_Bomb=0;
	let ItemNam_Item=0;

	@Initialize
	{
		SetLife(40);//
		LoadSE(ShotSE);
		LoadGraphic(imgEnemy);
		SetTexture(imgEnemy);
		alternative(GetArgument)
		case(0,1){Initialize_Fairy(4,100,100,255);}	//青妖精
		case(2,3){Initialize_Fairy(4,255,100,100);}	//赤妖精
	}
	
	@MainLoop
	{
		let tdif = GetCommonDataEx("title","Diffi");//本編の難易度判定("easy"〜"lunatic")
		
		if(count>=50&&count<=200&&!enmfin){
			alternative(GetArgument)
			case(0,1){
				if(count%20==0){
					PlaySE(ShotSE);
					alternative(tdif)
					case("easy"){
						CreateShot01(GetX,GetY,1,GetAngleToPlayer,BLUE01,10);
					}
					case("normal"){
						CreateShot01(GetX,GetY,3,GetAngleToPlayer,BLUE01,10);
					}
					case("hard"){
						CreateShot01(GetX,GetY,4,GetAngleToPlayer,BLUE01,10);
					}
					case("lunatic"){
						CreateShot01(GetX,GetY,3,GetAngleToPlayer,BLUE01,10);
						CreateShot01(GetX,GetY,4,GetAngleToPlayer,BLUE01,10);
						CreateShot01(GetX,GetY,5,GetAngleToPlayer,BLUE01,10);
					}
				}
			}
			case(2,3){
				if(count%20==0){
					PlaySE(ShotSE);
					alternative(tdif)
					case("easy"){
						CreateShot01(GetX,GetY,1,GetAngleToPlayer,RED01,10);
					}
					case("normal"){
						CreateShot01(GetX,GetY,3,GetAngleToPlayer,RED01,10);
					}
					case("hard"){
						CreateShot01(GetX,GetY,4,GetAngleToPlayer,RED01,10);
					}
					case("lunatic"){
						CreateShot01(GetX,GetY,3,GetAngleToPlayer,RED01,10);
						CreateShot01(GetX,GetY,4,GetAngleToPlayer,RED01,10);
						CreateShot01(GetX,GetY,5,GetAngleToPlayer,RED01,10);
					}
				}
			}
		}

		if(count==0){
			alternative(GetArgument)
			case(0,2){
				SetMovePosition02(GetX,MaxY+150,360)
			}
			case(1,3){
				SetMovePosition02(GetX,MaxY+150,360)
			}
		}
		if(GetX<MinX-100&&!enmfin){
		VanishEnemy;
		}
		if(GetX>MaxX+100&&!enmfin){
		VanishEnemy;
		}
		if(GetY<MinY-100&&!enmfin){
		VanishEnemy;
		}
		if(GetY>MaxY+100&&!enmfin){
		VanishEnemy;
		}
		SetCollisionA(GetX(),GetY(),16);//当たり判定(自弾)登録
		count++;
		yield;
	}
	
	@Finalize
	{//読みこんだファイルを削除

		loop(ItemNam_1up) {
			CreateItem(ITEM_1UP,GetX(),GetY());
		}
		loop(ItemNam_Bomb) {
			CreateItem(ITEM_BOMB,GetX(),GetY());
		}
		if (GetLife()<=0) {
			ascent (i in 0..ItemNam_Item) {
				CreateItem(ITEM_SCORE,GetX()+rand(-i*10,i*10),GetY()+rand(-i*10,i*10));
			}
		}
	}
	
	@DrawLoop
	{//描画
		DrawFairy(imgEnemy);
	}
#include_function".\lib_anime_fairy.txt"
}
